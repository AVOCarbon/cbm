CREATE VIEW 
	FI-D7_Sales
AS 
	SELECT
		LIGNE.GL_CODEARTICLE AS Intenal_reference,
		LIGNE.GL_TIERS AS Customer_code,
		LIGNE.GL_NUMERO AS InvoiceNumber,
		PIECE.GP_TOTALHT AS Value_in_currency,
		PIECE.GP_DEVISE AS Currency_code,
		PIECE.GP_TOTALQTESTOCK AS Qty,
		PIECE.GP_DATECREATION AS Selling_date


	FROM 
		LIGNE
			INNER JOIN PIECE 
				ON
				PIECE.GP_NUMERO = LIGNE.GL_NUMERO
	WHERE 
		LIGNE.GL_NATUREPIECEG = 'BLC'

;
